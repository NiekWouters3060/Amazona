from js import document # Tooltip-element tooltip = document.getElementById('tooltip') # Voorbeeld hotspots hotspots = [ {'x': 100, 'y': 150, 'text': 'Dit is een hotspot!'}, {'x': 300, 'y': 250, 'text': 'Hier is iets interessants!'} ] # Functie om de tooltip te tonen def show_tooltip(x, y, text): tooltip.style.left = f"{x}px" tooltip.style.top = f"{y}px" tooltip.textContent = text tooltip.style.display = 'block' # Functie om de tooltip te verbergen def hide_tooltip(): tooltip.style.display = 'none' # Mousemove event handler def on_mouse_move(event): x = event.clientX - event.target.getBoundingClientRect().left y = event.clientY - event.target.getBoundingClientRect().top found = False for hotspot in hotspots: distance = ((hotspot['x'] - x) ** 2 + (hotspot['y'] - y) ** 2) ** 0.5 if distance < 30: # Drempelwaarde show_tooltip(hotspot['x'], hotspot['y'], hotspot['text']) found = True break if not found: hide_tooltip() # Event listeners toevoegen image_container = document.getElementById('image-container') # Handlers binden zonder create_proxy def bind_events(): image_container.addEventListener('mousemove', on_mouse_move) image_container.addEventListener('mouseleave', hide_tooltip) # Bind de events bind_events()
